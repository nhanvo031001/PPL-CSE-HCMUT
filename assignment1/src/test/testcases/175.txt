
            string calculate(long long int n)
            {
                auto memo = std::vector<std::string>(n + 1);
                memo[0] = "0";
                memo[1] = "1";
                for(long long int i = 2; i <= n; ++i)
                {
                    if(i%2 == 0) memo[i] = memo[i/2] + "0" + memo[i/2];
                    else memo[i] = memo[i/2] + "1" + memo[i/2];
                }

                return memo[n];
            }
